%% example: \mygraphics{0.7\textwidth}{rolling_grandma}{This is my grandmother on inlinescates}{grandma}
%% requires: package graphicx
%% provides: including centered pictures/graphics with a boldfaced caption below

\newcommand{\mygraphics}[4]{
	\begin{figure}[h!]
		\centering
		\includegraphics[width=#1, keepaspectratio=true]{#2}
		\caption{#3}
		\label{fig:#4}
	\end{figure}
}
